<script>
  const magnifier = document.querySelector('.magnifier');
  const magnifiedText = document.querySelector('#magnifiedText');
  const textContainer = document.querySelector('#textContainer');

  document.addEventListener('mousemove', (e) => {
    const rect = textContainer.getBoundingClientRect();
    const x = e.clientX - rect.left;
    const y = e.clientY - rect.top;

    // Move magnifier
    magnifier.style.left = `${x}px`;
    magnifier.style.top = `${y}px`;

    // Adjust the magnified text to match the cursor movement
    magnifiedText.style.left = `${-x * 1.8 + 60}px`;  // Adjust for scaling effect
    magnifiedText.style.top = `${-y * 2 + 60}px`;
  });
</script>
